/addGoods.do   添加商品

```

method: get

request:
{
    categoryInfo: {  // 类目信息，类目枚举列表见文档 "商品类目约定.md" 及 catogary.json
        category1: 1,
        category2: 101,
        category3: 10101,
        category4: 1010101
    },
    basicInfo: { //商品基本信息
        title: "", // 商品标题
        subTitle: "", // 商品副标题
        authorString: "", //悦选小编说
        sellPrice: "99", //商品售价
        originalPrice: "99", //商品原价
        costPrice: "99" // 成本价
    },
    propertyInfo: { // 商品规格属性信息，根据商品的类目传不同的key value键值对，见所有规格列表接口
        "key1": "value1",
        "key2": "value2",
        ...
        "other": "" // 所有类目都有一个"其他"项
    },
    picInfo: { // 商品图片信息，所有的图片，都直接回传CDN返回的地址。
        mainPic: "", // 主图
        carouselPics: ['', ''], // 轮播图列表
        detailPics: ['', ''] // 详情页图片，注意按数组顺序，前面的在前面显示。
    },
    isPublish: 0   // 0: 仅添加商品到数据库  1:添加商品，同时发布到线上
}

response :
{
  status: {
    code: 1001,  //所有的成功都是1001
    msg: "添加成功！"
  },
  data: {
  }
}
```
