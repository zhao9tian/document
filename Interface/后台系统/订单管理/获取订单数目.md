/getOrderNum.do 获取订单数目

请求方式: get
```
request:{
}
response : {
    status : {
        code : 1001,//返回状态
        msg:"请求成功" //返回信息
    },
    data: {
        refunding: 1 // 退款中订单数目
        waitDelivery: 1 // 等待发货的订单数目

        awaitDelivery : 1     //待发货订单数目
        takeGoods : 1   //待收货订单数目
        closeOrder ：1      //关闭订单数目
    }
}
```
